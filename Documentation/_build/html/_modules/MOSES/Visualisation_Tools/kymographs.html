[docs]def kymograph_img(vid, pos=None, axis=1, proj_fn=None):
    import numpy as np 
    
    if pos is None:    
        if proj_fn is None:
            return np.mean(vid, axis=axis)
        else:
            return proj_fn(vid, axis=axis)
    else:
        if axis == 0:
            return vid[pos]
        if axis == 1:
            return vid[:,pos]
        if axis == 2:
            return vid[:,:,pos]
